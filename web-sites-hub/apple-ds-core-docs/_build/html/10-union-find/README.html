        self.parent = list(range(n))
        self.rank = [0] * n
        self.count = n  # è¿é€šåˆ†é‡æ•°

def num_islands(grid):
    if not grid:
        return 0
    
    m, n = len(grid), len(grid[0])
    uf = UnionFind(m * n)
    
    for i in range(m):
        for j in range(n):
