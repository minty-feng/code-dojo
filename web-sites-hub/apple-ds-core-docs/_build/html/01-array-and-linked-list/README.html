def find_middle(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next

    slow = fast = head

        slow = slow.next
        fast = fast.next.next
        if slow == fast:

    return False

def reverse_list(head):
    prev = None
    curr = head
    while curr:
        next_node = curr.next
        curr.next = prev
        prev = curr
        curr = next_node
    return prev

200        if (head == nullptr) {
201            head = tail = newNode;
202        } else {
203            newNode->next = head;
204            head->prev = newNode;
205            head = newNode;

208    }
